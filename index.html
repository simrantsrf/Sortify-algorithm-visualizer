Sorting Algorithms

Bubble Sort

for i from 0 to n-1
 for j from 0 to n-i-1
  if arr[j] > arr[j+1]
   swap(arr[j], arr[j+1])

Selection Sort

for i from 0 to n-1
 min = i
 for j from i+1 to n
  if arr[j] < arr[min]
   min = j
 swap(arr[i], arr[min])

Insertion Sort

for i from 1 to n
 key = arr[i]
 j = i-1
 while j>=0 and arr[j]>key
  arr[j+1]=arr[j]
  j--
 arr[j+1]=key

Searching Algorithms

Binary Search

while low <= high
 mid = (low+high)/2
 if target == arr[mid]
  return mid
 else if target > arr[mid]
  low = mid+1
 else
  high = mid-1

Graph Algorithms

DFS

DFS(node):
 mark node visited
 for each neighbor
  if not visited
   DFS(neighbor)

BFS

BFS(start):
 create queue
 mark start visited
 push start
 while queue not empty
  node = pop
  visit neighbors

Dijkstra

initialize distance
while nodes remain
 pick smallest distance
 update neighbors
